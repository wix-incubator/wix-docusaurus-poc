import Operation from '@site/src/components/ContentRenderer/OpenApi/Operation';
const source = `wix.ecom.v1.cart`;
const operationId = `com.wix.ecom.cart.api.v1.CartService.GetCart`;

<!---
<Operation source={source} operationId={operationId}/>
-->


import { useState, useEffect } from 'react';
import axios from 'axios';
const opeartionMock = require('@site/mocks/create-cart.json');
export const OperationWrapper = () => {
  const [operation, setOperation] = useState();
    useEffect(() => {
    axios.get(`https://fakestoreapi.com/products?source=${source}&operationId=${operationId}`).then(res => {
      if (res.status === 200) {
        setOperation(opeartionMock)
      }
    })
  }, [])
  if (!operation) {
    return <div>Fetching Operation...</div>
  }else{
    return <Operation operation={operation}/>
  }
}

<OperationWrapper />
